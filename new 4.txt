#REQUISITOS DO TRABALHO:
 Possibilidade de visualizar os filmes de um determinado gênero = OK
 Possibilidade de visualizar os detalhes de cada filme = OK
 Possibilidade de buscar um filme por palavra-chave = OK
 Possibilidade de visualizar os filmes e séries que já foram assistidos
	-> SERVIÇO DE FILME = OK
	-> SERVIÇO DE SERIE = OK

 Possibilidade de votar nos filmes que mais gostei = OK
 Possibilidade de marcar um filme ou série para ser visto no futuro:
	-> SERVIÇO DE FILME = OK
	-> SERVIÇO DE SERIE = OK

 Possibilidade de exibir os filmes mais vistos por categorias = OK;
 Possibilidade de abrir um chamado técnico de algum problema que está acontecendo = OK;

##CREATE REDE NO DOCKER
docker network create 76aoj-netflix-net

##BAIXAR IMAGE DO DOCKER HUB
docker pull postgres:12-alpine

##CREATE/EXECUTAR CONTAINER DOCKER
-> BANCOS DE DADOS:
	docker run -p 5432:5432 --name netflix-category-pg12  --network 76aoj-netflix-net -e POSTGRES_PASSWORD=root -e POSTGRES_DB=db_netflix_category  postgres:12-alpine
	docker run -p 5433:5432 --name netflix-movies-pg12    --network 76aoj-netflix-net -e POSTGRES_PASSWORD=root -e POSTGRES_DB=db_netflix_movies    postgres:12-alpine
	docker run -p 5434:5432 --name netflix-series-pg12    --network 76aoj-netflix-net -e POSTGRES_PASSWORD=root -e POSTGRES_DB=db_netflix_series    postgres:12-alpine
	docker run -p 5435:5432 --name netflix-user-pg12      --network 76aoj-netflix-net -e POSTGRES_PASSWORD=root -e POSTGRES_DB=db_netflix_user      postgres:12-alpine
	docker run -p 5436:5432 --name netflix-help-desk-pg12 --network 76aoj-netflix-net -e POSTGRES_PASSWORD=root -e POSTGRES_DB=db_netflix-help-desk postgres:12-alpine
